//Este código configura el módulo Multer en NestJS para controlar cómo y dónde 
// se guardan los archivos que suben los usuarios a tu servidor.

import { diskStorage } from 'multer';

MulterModule.register({
    
  // 1. Define el tipo de almacenamiento. 'diskStorage' significa que los archivos se guardarán en el disco duro del servidor.
  storage: diskStorage({
    
    // 2. Define la carpeta de destino. En este caso, crea o usa una carpeta llamada 'uploads' en la raíz del proyecto.
    destination: './uploads',
    
    // 3. Define una función para cambiar el nombre del archivo antes de guardarlo.
    filename: (req, file, cb) => {
      // Genera un nombre único usando los milisegundos actuales (Date.now()) unidos al nombre original del archivo.
      // Esto evita que un archivo nuevo borre a otro que se llame igual.
      const filename = `${Date.now()}-${file.originalname}`;
      
      // Ejecuta el callback enviando 'null' (indica que no hay error) y el nuevo nombre del archivo.
      cb(null, filename);
    },
  }),
});
